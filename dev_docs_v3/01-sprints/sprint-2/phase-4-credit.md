```typescript
@OnEvent('invoice.past_due')
async handleInvoicePastDue(payload: { companyId: string, tenantId: string, daysPastDue: number }) {
  if (payload.daysPastDue >= 60) {
    const existingHold = await this.prisma.creditHold.findFirst({
      where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE', type: 'PAST_DUE' },
    });
    if (!existingHold) {
      await this.create({
        companyId: payload.companyId,
        tenantId: payload.tenantId,
        type: 'PAST_DUE',
        reason: `Invoice 60+ days past due`,
        autoGenerated: true,
      });
    }
  }
}

@OnEvent('invoice.created')
async handleNewInvoice(payload: { companyId: string, tenantId: string }) {
  const limit = await this.prisma.creditLimit.findFirst({
    where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE' },
  });
  if (limit && limit.usedCredit > limit.creditLimit * 1.1) {
    const existingHold = await this.prisma.creditHold.findFirst({
      where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE', type: 'OVER_LIMIT' },
    });
    if (!existingHold) {
      await this.create({
        companyId: payload.companyId,
        tenantId: payload.tenantId,
        type: 'OVER_LIMIT',
        reason: `Balance exceeds credit limit by 10%+`,
        autoGenerated: true,
      });
    }
  }
}

@OnEvent('payment.received')
async handlePaymentReceived(payload: { companyId: string, tenantId: string }) {
  // Check if hold conditions are now resolved
  const limit = await this.prisma.creditLimit.findFirst({
    where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE' },
  });

  // Auto-release over-limit hold if now under limit
  if (limit && limit.usedCredit <= limit.creditLimit) {
    const overLimitHold = await this.prisma.creditHold.findFirst({
      where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE', type: 'OVER_LIMIT' },
    });
    if (overLimitHold) {
      await this.release(overLimitHold.id, payload.tenantId, 'system', 'Auto-released: balance under limit');
    }
  }

  // Check if all invoices now current (< 60 days)
  const overdueInvoices = await this.prisma.invoice.count({
    where: {
      companyId: payload.companyId,
      tenantId: payload.tenantId,
      status: 'OVERDUE',
      dueDate: { lt: new Date(Date.now() - 60 * 24 * 60 * 60 * 1000) },
    },
  });
  if (overdueInvoices === 0) {
    const pastDueHold = await this.prisma.creditHold.findFirst({
      where: { companyId: payload.companyId, tenantId: payload.tenantId, status: 'ACTIVE', type: 'PAST_DUE' },
    });
    if (pastDueHold) {
      await this.release(pastDueHold.id, payload.tenantId, 'system', 'Auto-released: no invoices 60+ days past due');
    }
  }
}
```

```typescript
async checkCreditAvailable(companyId: string, tenantId: string, orderAmount: number): Promise<CreditCheckResult> {
  // Check for active holds
  const activeHold = await this.prisma.creditHold.findFirst({
    where: { companyId, tenantId, status: 'ACTIVE' },
  });
  if (activeHold) {
    return { available: false, onHold: true, holdReason: activeHold.reason, holdType: activeHold.type };
  }

  // Check credit limit
  const limit = await this.prisma.creditLimit.findFirst({
    where: { companyId, tenantId, status: 'ACTIVE' },
  });
  if (!limit) {
    return { available: true, onHold: false, noLimit: true }; // No limit = no restriction
  }

  const remainingCredit = limit.creditLimit - limit.usedCredit;
  if (remainingCredit < orderAmount) {
    return {
      available: false,
      onHold: false,
      creditLimit: limit.creditLimit,
      usedCredit: limit.usedCredit,
      remainingCredit,
      requestedAmount: orderAmount,
    };
  }

  return { available: true, onHold: false, remainingCredit };
}
```

```typescript
@Get('dashboard')
async getDashboard(@CurrentTenant() tenantId: string) {
  const [totalExposure, totalLimit, activeHolds, pendingApps, overdueAccounts] = await Promise.all([
    this.prisma.creditLimit.aggregate({
      where: { tenantId, status: 'ACTIVE' },
      _sum: { usedCredit: true },
    }),
    this.prisma.creditLimit.aggregate({
      where: { tenantId, status: 'ACTIVE' },
      _sum: { creditLimit: true },
    }),
    this.prisma.creditHold.count({
      where: { tenantId, status: 'ACTIVE' },
    }),
    this.prisma.creditApplication.count({
      where: { tenantId, status: 'PENDING' },
    }),
    this.prisma.creditLimit.findMany({
      where: {
        tenantId,
        status: 'ACTIVE',
        usedCredit: { gt: 0 },
      },
      include: { company: { select: { id: true, name: true } } },
      orderBy: { usedCredit: 'desc' },
      take: 10,
    }),
  ]);

  const totalExposureValue = totalExposure._sum.usedCredit || 0;
  const totalLimitValue = totalLimit._sum.creditLimit || 0;
  const utilization = totalLimitValue > 0 ? (totalExposureValue / totalLimitValue) * 100 : 0;

  return {
    totalExposure: totalExposureValue,
    totalAvailable: totalLimitValue - totalExposureValue,
    totalLimit: totalLimitValue,
    creditUtilization: Math.round(utilization * 100) / 100,
    activeHolds,
    pendingApplications: pendingApps,
    atRiskAccounts: overdueAccounts.map(a => ({
      companyId: a.company.id,
      companyName: a.company.name,
      usedCredit: a.usedCredit,
      creditLimit: a.creditLimit,
      utilization: Math.round((a.usedCredit / a.creditLimit) * 100),
    })),
  };
}
```

```typescript
async writeOff(collectionId: string, amount: number, reason: string, tenantId: string, userId: string) {
  const collection = await this.prisma.collectionItem.findFirst({
    where: { id: collectionId, tenantId },
  });
  if (!collection) throw new NotFoundException();

  // Update collection status
  await this.prisma.collectionItem.update({
    where: { id: collectionId },
    data: {
      status: 'WRITTEN_OFF',
      writeOffAmount: amount,
      writeOffReason: reason,
      writeOffDate: new Date(),
      writeOffById: userId,
    },
  });

  // Reduce used credit on the company's credit limit
  await this.prisma.creditLimit.updateMany({
    where: { companyId: collection.companyId, tenantId },
    data: { usedCredit: { decrement: amount } },
  });

  // Log activity
  await this.logActivity(collectionId, tenantId, userId, 'WRITE_OFF', `Written off $${amount}: ${reason}`);

  // Emit event for accounting
  this.eventEmitter.emit('credit.writeoff', {
    collectionId, companyId: collection.companyId, amount, tenantId,
  });

  return collection;
}
```

#### CRED-001e: Add credit score calculation (1-2h)
**Create:** `apps/api/src/modules/credit/credit-score.service.ts`

```typescript
@Injectable()
export class CreditScoreService {
  async calculateScore(companyId: string, tenantId: string): Promise<CreditScore> {
    const [paymentHistory, accountAge, utilization, orderVolume, claimsHistory] = await Promise.all([
      this.getPaymentHistoryScore(companyId, tenantId),    // 40%
      this.getAccountAgeScore(companyId, tenantId),         // 20%
      this.getUtilizationScore(companyId, tenantId),        // 20%
      this.getOrderVolumeScore(companyId, tenantId),        // 10%
      this.getClaimsHistoryScore(companyId, tenantId),      // 10%
    ]);

    const totalScore = Math.round(
      paymentHistory * 0.4 +
      accountAge * 0.2 +
      utilization * 0.2 +
      orderVolume * 0.1 +
      claimsHistory * 0.1
    );

    const tier = totalScore >= 800 ? 'EXCELLENT'
      : totalScore >= 700 ? 'GOOD'
      : totalScore >= 600 ? 'FAIR'
      : totalScore >= 500 ? 'POOR'
      : 'VERY_POOR';

    return {
      companyId,
      totalScore,
      tier,
      components: { paymentHistory, accountAge, utilization, orderVolume, claimsHistory },
      calculatedAt: new Date(),
    };
  }

  private async getPaymentHistoryScore(companyId: string, tenantId: string): Promise<number> {
    // 850 = all on time, -50 per late payment, -100 per 60+ day late
    const payments = await this.prisma.payment.findMany({
      where: { companyId, tenantId },
      include: { invoice: true },
    });

    let score = 850;
    for (const payment of payments) {
      if (!payment.invoice?.dueDate) continue;
      const daysLate = Math.floor((payment.paidAt.getTime() - payment.invoice.dueDate.getTime()) / (1000 * 60 * 60 * 24));
      if (daysLate > 60) score -= 100;
      else if (daysLate > 30) score -= 50;
      else if (daysLate > 0) score -= 25;
    }

    return Math.max(300, Math.min(850, score));
  }

  // Similar implementations for other components...
}
```
